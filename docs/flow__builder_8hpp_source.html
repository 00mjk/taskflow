561 template <typename C>
562 Task FlowBuilder::emplace(C&& c) {
563  Node* n;
564  if constexpr(is_static_task_v<C>) {
565  n = _graph.emplace_back(
566  std::in_place_type_t<Node::StaticWork>{}, std::forward<C>(c)
567  );
568  }
569  else if constexpr(is_dynamic_task_v<C>) {
570  n = _graph.emplace_back(
571  std::in_place_type_t<Node::DynamicWork>{}, std::forward<C>(c)
572  );
573  }
574  else if constexpr(is_condition_task_v<C>) {
575  n = _graph.emplace_back(
576  std::in_place_type_t<Node::ConditionWork>{}, std::forward<C>(c)
577  );
578  }
579 #ifdef TF_ENABLE_CUDA
580  else if constexpr(is_cudaflow_task_v<C>) {
581  n = _graph.emplace_back(
582  std::in_place_type_t<Node::cudaFlowWork>{}, std::forward<C>(c)
583  );
584  }
585 #endif
586  else {
587  static_assert(dependent_false_v<C>, "invalid task callable");
588  }
589 
590  return Task(n);
591 }

616 
617  auto itr = keys.begin();
618  auto end = keys.end();

621  return;
